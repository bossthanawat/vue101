# Data

คือค่า variable ที่เราสามารถนำไปใช้

<script>
  export default {
    data() {
      return {
        weapon: 'knife',
        series: 'Z',
        price: 300,
        colors: ['red', 'green', 'yellow']
      }
    }
  }
</script>

เราสามารถใช้ {{}} สำหรับการนำ data variable ไปใส่ได้

<template>
  <div>
    <p>weapon: {{ weapon }}</p>
    <p>series: {{ series }}</p>
    <p>price: {{ price }}</p>
    <p>colors: {{ colors }}</p>
  </div>
</template>

<script>
  export default {
    data() {
      return {
        weapon: 'knife',
        series: 'Z',
        price: 300,
        colors: ['red', 'green', 'yellow']
    }
  }
}
</script>

แสดงผล

weapon: knife

series: Z

price: 300

colors: [ "red", "green", "yellow" ]

TIP

ภายใต้ {{}}สามารถเขียนพวก check true false แบบง่ายๆ เช่น {{ price > 300 ?'แพงจัง':'ถูกมาก' }}

# Computed

ใช้ในเวลาที่เราต้องการคำนวณค่า data ก่อนนำไปใช้อีกทีนึง โดยที่ Computed property จะเป็น method ที่ไม่มี argument ตัว Computed property จะอัพเดตให้เองเองเมื่อข้อมูลที่สนใจมีการเปลี่ยนแปลง หรือจะเรียกมันว่า getter method ก็ว่าได้

<template>
  <div>
    <p>fullname: {{ fullname }}</p>
    <p>price: {{ price }}</p>
    <p>eventPrice: {{ eventPrice }}</p>
    <p>colors: {{ colors }}</p>
  </div>
</template>

<script>
  export default {
    data() {
      return {
        weapon: 'knife',
        series: 'Z',
        price: 300,
        colors: ['red', 'green', 'yellow']
      },
    },
    computed: {
      fullname() {
        return this.weapon + ' ' + this.series
      },
      eventPrice() {
        if(this.price > 299) return this.price * 2
        else return this.price
      }
    }
  }
</script>

แสดงผล

fullname: knife Z

price: 300

eventPrice: 600

colors: [ "red", "green", "yellow" ]

# Methods

เป็นส่วนสำหรับเขียน methods เพื่อเอาไปใช้

<template>
  <div>
    ...
    <p>price: {{ price }}</p>
    ...
    <button @click="increasePrice">Increase Price</button>
    <button @click="decreasePrice">Decrease Price</button>
  </div>
</template>
<script>
  export default {
    ...
    methods: {
      increasePrice() {
        this.price++
      },
      decreasePrice() {
        this.price--
      }
    }
  }
</script>

@click="..." ใช้สำหรับดักจับ event ต่างๆ อยู่ในหัวข้อ Directive v-on

แสดงผล

fullname: knife Z

price: 300

eventPrice: 600

colors: [ "red", "green", "yellow" ]

# Watch

เป็นตัวช่วย trigger data ถ้ามีการเปลี่ยนแปลงเกิดขึ้น

<script>
  export default {
    ...
    watch: {
      price(val, oldVal) {
        console.log("price : " + val, "|", "oldPrice : " + oldVal);; //จะlog ทุกครั้งที price มีการเปลี่ยนแปลง
      }
    }
  }
</script>

ถ้าใช้ chrome browser ให้กด F12 ดูที่ tab "Console" เราจะเห็น log จาก console.log ที่นั้น เรายังสามารถ log ลง Console ได้ด้วยรูปแบบอื่นๆได้อีก ว่าด้วยเรื่อง Consoleใน JavaScript (opens new window)

  • beforeCreate
  • created
  • beforeMount
  • mounted
  • beforeUpdate
  • updated
  • beforeDestroy
  • destoryed

  export default {
    ...
    created() {

      //Ex. get api เพื่อเอาข้อมูลมาใช้หรือโชว์ ก่อนที่ component จะเริ่ม render

  </div>
</template>

# Emit

เป็นการส่งค่าออกจาก component

ตัวอย่างจาก khomkrit (opens new window)

<!-- file Create.vue -->
<template>
  <form @submit.prevent="saveColor"> <!-- submit แล้วจะเรียก saveColor-->
    <div>
      <label>Input new Color</label>
      <input
        type="text"
        class="form-control"
        placeholder="input"
        v-model="color"
      />
    </div>
    <button type="submit">Submit</button>
  </form>
</template>
<script>
export default {
  data() {
    return {
      color: ''
    }
  },
  methods: {
    saveColor() {
      this.$emit('createColor', this.color) // ประกาศ emit event ชื่อ 'createColor' ที่ส่งค่า color
    }
  }
}
</script>

<template>
  <div>
    <app-create @createColor="addColor"></app-create>
     <!-- ใช้ v-on รับ 'createColor' event emit ที่ประกาศไว้ในตัว AppCreate และต้องรับด้วย methods เท่านั้น -->
     {{colors}}
  </div>
</template>
<script>
  import Create from './Create.vue'
  export default {
    data() { 
      return {
        colors:[]
      }
    },
    methods: {
      addColor(color) {
        this.colors.push(color) 
        // ค่าของ color คือ this.color จาก AppCreate ที่ส่งมา
        // มันก็เหมือนกับเขียนรับ event @click @input เพียงแต่อันนี้เราเขียน event เอง ตั้งชื่อเอง return ค่าเอง
      }
    },
    components: {
      AppCreate: Create
    }
  }
</script>
